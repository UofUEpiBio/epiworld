206 template<typename ModelType>
207 inline std::vector<epiworld_double> tests_calculate_avg_transitions(
208  const std::vector<std::vector<epiworld_double>>& transitions,
209  const ModelType & model
210 ) {
211  if (transitions.empty())
212  return {};
213 
214  // Creating an average across the transitions vectors
215  std::vector<epiworld_double> avg_transitions(transitions[0].size(), 0.0);
216  for (size_t i = 0; i < transitions.size(); ++i)
217  {
218  for (size_t j = 0; j < transitions[i].size(); ++j)
219  {
220  avg_transitions[j] += transitions[i][j];
221  }
222  }
223 
224  // Normalizing the average
225  auto n_states = model.get_n_states();
226  for (size_t i = 0; i < n_states; ++i)
227  {
228  double rowsums = 0.0;
229  for (size_t j = 0; j < n_states; ++j)
230  {
231  rowsums += avg_transitions[j * n_states + i];
232  }
233 
234  // Normalizing the rows
235  // If the row is empty, we skip it
236  if (rowsums == 0.0)
237  continue;
238 
239  for (size_t j = 0; j < n_states; ++j)
240  {
241  avg_transitions[j * n_states + i] /= rowsums;
242  }
243  }
244 
245  return avg_transitions;
246 };
