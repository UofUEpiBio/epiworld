208  // Exposure coefficient
209  const double coef_exposure = _m->coefs_infect[0u];
210 
211  // This computes the prob of getting any neighbor variant
212  size_t nviruses_tmp = 0u;
213 
214  double baseline = 0.0;
215  for (size_t k = 0u; k < _m->coef_infect_cols.size(); ++k)
216  baseline += p->operator[](k) * _m->coefs_infect[k + 1u];
217 
218  for (auto & neighbor: p->get_neighbors())
219  {
220 
221  if (neighbor->get_virus() == nullptr)
222  continue;
223 
224  auto & v = neighbor->get_virus();
225 
226  #ifdef EPI_DEBUG
227  if (nviruses_tmp >= m->array_virus_tmp.size())
228  throw std::logic_error("Trying to add an extra element to a temporal array outside of the range.");
229  #endif
230 
231  /* And it is a function of susceptibility_reduction as well */
232  m->array_double_tmp[nviruses_tmp] =
233  baseline +
234  (1.0 - p->get_susceptibility_reduction(v, m)) *
235  v->get_prob_infecting(m) *
236  (1.0 - neighbor->get_transmission_reduction(v, m)) *
237  coef_exposure

246  }

248  // No virus to compute
