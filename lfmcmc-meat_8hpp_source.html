40  epiworld_double scale,
41  epiworld_double lb,
42  epiworld_double ub
43 ) {
44 
45  LFMCMCProposalFun<TData> fun =
46  [scale,lb,ub](
47  std::vector< epiworld_double >& new_params,
48  const std::vector< epiworld_double >& old_params,
49  LFMCMC<TData>* m
50  ) {
51 
52  // Making the proposal
53  for (size_t p = 0u; p < m->get_n_params(); ++p)
54  new_params[p] = old_params[p] + m->rnorm() * scale;
55 
56  // Checking boundaries
57  epiworld_double d = ub - lb;
58  int odd;
59  epiworld_double d_above, d_below;
60  for (auto & p : new_params)
61  {
62 
63  // Correcting if parameter goes above the upper bound
64  if (p > ub)
65  {
66  d_above = p - ub;
67  odd = static_cast<int>(std::floor(d_above / d)) % 2;
68  d_above = d_above - std::floor(d_above / d) * d;
69 
70  p = (lb + d_above) * odd +
71  (ub - d_above) * (1 - odd);
72 
73  // Correcting if parameter goes below upper bound
74  } else if (p < lb)
75  {
76  d_below = lb - p;
77  int odd = static_cast<int>(std::floor(d_below / d)) % 2;
78  d_below = d_below - std::floor(d_below / d) * d;
79 
80  p = (ub - d_below) * odd +
81  (lb + d_below) * (1 - odd);
82  }
83 
84  }
85 
86  #ifdef EPI_DEBUG
87  for (auto & p : new_params)
88  if (p < lb || p > ub)
89  throw std::range_error("The parameter is out of bounds.");
90  #endif
91 
92 
93  return;
94 
95  };
