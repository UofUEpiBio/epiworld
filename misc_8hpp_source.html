127 template<typename TSeq = EPI_DEFAULT_TSEQ, typename TDbl = epiworld_double >
128 inline int roulette(
129  const std::vector< TDbl > & probs,
130  Model<TSeq> * m
131  )
132 {
133 
134  // Step 1: Computing the prob on none
135  TDbl p_none = 1.0;
136  std::vector< int > certain_infection;
137  certain_infection.reserve(probs.size());
138 
139  for (epiworld_fast_uint p = 0u; p < probs.size(); ++p)
140  {
141  p_none *= (1.0 - probs[p]);
142 
143  if (probs[p] > (1 - 1e-100))
144  certain_infection.push_back(p);
145 
146  }
147 
148  TDbl r = static_cast<TDbl>(m->runif());
149  // If there are one or more probs that go close to 1, sample
150  // uniformly
151  if (certain_infection.size() > 0)
152  return certain_infection[std::floor(r * certain_infection.size())];
153 
154  // Step 2: Calculating the prob of none or single
155  std::vector< TDbl > probs_only_p(probs.size());
156  TDbl p_none_or_single = p_none;
157  for (epiworld_fast_uint p = 0u; p < probs.size(); ++p)
158  {
159  probs_only_p[p] = probs[p] * (p_none / (1.0 - probs[p]));
160  p_none_or_single += probs_only_p[p];
161  }
162 
163  // Step 3: Roulette
164  TDbl cumsum = p_none/p_none_or_single;
165  if (r < cumsum)
166  {
167  return -1;
168  }
169 
170  for (epiworld_fast_uint p = 0u; p < probs.size(); ++p)
171  {
172  // If it yield here, then bingo, the individual will acquire the disease
173  cumsum += probs_only_p[p]/(p_none_or_single);
174  if (r < cumsum)
175  return static_cast<int>(p);
176 
177  }
178 
179 
180  #ifdef EPI_DEBUG
181  printf_epiworld("[epi-debug] roulette::cumsum = %.4f\n", cumsum);
182  #endif
183 
184  return static_cast<int>(probs.size() - 1u);
185 
186 }
