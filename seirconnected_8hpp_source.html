160  epiworld::UpdateFun<TSeq> update_susceptible = [](
161  epiworld::Agent<TSeq> * p, epiworld::Model<TSeq> * m
162  ) -> void
163  {
164 
165  // Sampling how many individuals
166  int ndraw = m->rbinom();
167 
168  if (ndraw == 0)
169  return;
170 
171  ModelSEIRCONN<TSeq> * model = dynamic_cast<ModelSEIRCONN<TSeq> *>(m);
172  size_t ninfected = model->get_n_infected();
173 
174  // Drawing from the set
175  int nviruses_tmp = 0;
176  for (int i = 0; i < ndraw; ++i)
177  {
178  // Now selecting who is transmitting the disease
179  int which = static_cast<int>(
180  std::floor(ninfected * m->runif())
181  );
182 
183  /* There is a bug in which runif() returns 1.0. It is rare, but
184  * we saw it here. See the Notes section in the C++ manual
185  * https://en.cppreference.com/mwiki/index.php?title=cpp/numeric/random/uniform_real_distribution&oldid=133329
186  * And the reported bug in GCC:
187  * https://gcc.gnu.org/bugzilla/show_bug.cgi?id=63176
188  *
189  */
190  if (which == static_cast<int>(ninfected))
191  --which;
192 
193  epiworld::Agent<TSeq> & neighbor = *model->infected[which];
194 
195  // Can't sample itself
196  if (neighbor.get_id() == p->get_id())
197  continue;
198 
199  // The neighbor is infected by construction
200  auto & v = neighbor.get_virus();
201 
202  #ifdef EPI_DEBUG
203  if (nviruses_tmp >= static_cast<int>(m->array_virus_tmp.size()))
204  throw std::logic_error("Trying to add an extra element to a temporal array outside of the range.");
205  #endif
206 
207  /* And it is a function of susceptibility_reduction as well */
208  m->array_double_tmp[nviruses_tmp] =
209  (1.0 - p->get_susceptibility_reduction(v, m)) *
210  v->get_prob_infecting(m) *
211  (1.0 - neighbor.get_transmission_reduction(v, m))
212  ;
213 
214  m->array_virus_tmp[nviruses_tmp++] = &(*v);
215 
216  }
217 
218  // No virus to compute
219  if (nviruses_tmp == 0u)
220  return;
221 
222  // Running the roulette
223  int which = roulette(nviruses_tmp, m);
224 
225  if (which < 0)
226  return;
227 
228  p->set_virus(
229  *m->array_virus_tmp[which],
230  m,
231  ModelSEIRCONN<TSeq>::EXPOSED
232  );
233 
234  return;

238  epiworld::UpdateFun<TSeq> update_infected = [](
239  epiworld::Agent<TSeq> * p, epiworld::Model<TSeq> * m
240  ) -> void {
241 
242  auto state = p->get_state();
243 
244  if (state == ModelSEIRCONN<TSeq>::EXPOSED)
245  {
246 
247  // Getting the virus
248  auto & v = p->get_virus();
249 
250  // Does the agent become infected?
251  if (m->runif() < 1.0/(v->get_incubation(m)))
252  {
253 
254  p->change_state(m, ModelSEIRCONN<TSeq>::INFECTED);
255  return;
256 
257  }
258 
259 
260  } else if (state == ModelSEIRCONN<TSeq>::INFECTED)
261  {
262 
263 
264  // Odd: Die, Even: Recover
265  epiworld_fast_uint n_events = 0u;
266  const auto & v = p->get_virus();
267 
268  // Recover
269  m->array_double_tmp[n_events++] =
270  1.0 - (1.0 - v->get_prob_recovery(m)) * (1.0 - p->get_recovery_enhancer(v, m));
271 
272  #ifdef EPI_DEBUG
273  if (n_events == 0u)
274  {
275  printf_epiworld(
276  "[epi-debug] agent %i has 0 possible events!!\n",
277  static_cast<int>(p->get_id())
278  );
279  throw std::logic_error("Zero events in exposed.");
280  }
281  #else
282  if (n_events == 0u)
283  return;
284  #endif
285 
286 
287  // Running the roulette
288  int which = roulette(n_events, m);
289 
290  if (which < 0)
291  return;
292 
293  // Which roulette happen?
294  p->rm_virus(m);
295 
296  return ;

298  } else
299  throw std::logic_error("This function can only be applied to exposed or infected individuals. (SEIR)") ;
