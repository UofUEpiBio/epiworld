284 template<typename TSeq>
285 inline size_t ModelSEIRMixingQuarantine<TSeq>::sample_agents(
286  epiworld::Agent<TSeq> * agent,
287  std::vector< size_t > & sampled_agents
288  )
289 {
290 
291  size_t agent_group_id = agent->get_entity(0u).get_id();
292  size_t ngroups = this->entities.size();
293 
294  int samp_id = 0;
295  for (size_t g = 0; g < ngroups; ++g)
296  {
297 
298  size_t group_size = n_infected_per_group[g];
299 
300  if (group_size == 0u)
301  continue;
302 
303  // How many from this entity?
304  int nsamples = epiworld::Model<TSeq>::rbinom(
305  group_size,
306  adjusted_contact_rate[g] * contact_matrix[
307  MM(agent_group_id, g, ngroups)
308  ]
309  );
310 
311  if (nsamples == 0)
312  continue;
313 
314  // Sampling from the entity
315  for (int s = 0; s < nsamples; ++s)
316  {
317 
318  // Randomly selecting an agent
319  int which = epiworld::Model<TSeq>::runif() * group_size;
320 
321  // Correcting overflow error
322  if (which >= static_cast<int>(group_size))
323  which = static_cast<int>(group_size) - 1;
324 
325  #ifdef EPI_DEBUG
326  auto & a = this->population.at(infected.at(entity_indices[g] + which));
327  #else
328  auto & a = this->get_agent(infected[entity_indices[g] + which]);
329  #endif
330 
331  #ifdef EPI_DEBUG
332  if (a.get_state() != ModelSEIRMixingQuarantine<TSeq>::INFECTED)
333  throw std::logic_error(
334  "The agent is not infected, but it should be."
335  );
336  #endif
337 
338  // Can't sample itself
339  if (a.get_id() == agent->get_id())
340  continue;
341 
342  sampled_agents[samp_id++] = a.get_id();
343 
344  }
345 
346  }
347 
348  return samp_id;
349

365 template<typename TSeq>
366 inline void ModelSEIRMixingQuarantine<TSeq>::reset()
367 {
368 
369  Model<TSeq>::reset();
370 
371  // Checking contact matrix's rows add to one
372  size_t nentities = this->entities.size();
373  if (this->contact_matrix.size() != nentities*nentities)
374  throw std::length_error(
375  std::string("The contact matrix must be a square matrix of size ") +
376  std::string("nentities x nentities. ") +
377  std::to_string(this->contact_matrix.size()) +
378  std::string(" != ") + std::to_string(nentities*nentities) +
379  std::string(".")
380  );
381 
382  for (size_t i = 0u; i < this->entities.size(); ++i)
383  {
384  double sum = 0.0;
385  for (size_t j = 0u; j < this->entities.size(); ++j)
386  {
387  if (this->contact_matrix[MM(i, j, nentities)] < 0.0)
388  throw std::range_error(
389  std::string("The contact matrix must be non-negative. ") +
390  std::to_string(this->contact_matrix[MM(i, j, nentities)]) +
391  std::string(" < 0.")
392  );
393  sum += this->contact_matrix[MM(i, j, nentities)];
394  }
395  if (sum < 0.999 || sum > 1.001)
396  throw std::range_error(
397  std::string("The contact matrix must have rows that add to one. ") +
398  std::to_string(sum) +
399  std::string(" != 1.")
400  );
401  }
402 
403  // Do it the first time only
404  sampled_agents.resize(Model<TSeq>::size());
405 
406  // We only do it once
407  n_infected_per_group.resize(this->entities.size(), 0u);
408  std::fill(n_infected_per_group.begin(), n_infected_per_group.end(), 0u);
409 
410  // We are assuming one agent per entity
411  infected.resize(Model<TSeq>::size());
412  std::fill(infected.begin(), infected.end(), 0u);
413 
414  // This will say when do the groups start in the `infected` vector
415  entity_indices.resize(this->entities.size(), 0u);
416  std::fill(entity_indices.begin(), entity_indices.end(), 0u);
417  for (size_t i = 1u; i < this->entities.size(); ++i)
418  {
419 
420  entity_indices[i] +=
421  this->entities[i - 1].size() +
422  entity_indices[i - 1]
423  ;
424 
425  }
426 
427  // Adjusting contact rate
428  adjusted_contact_rate.clear();
429  adjusted_contact_rate.resize(this->entities.size(), 0.0);
430 
431  for (size_t i = 0u; i < this->entities.size(); ++i)
432  {
433 
434  adjusted_contact_rate[i] =
435  Model<TSeq>::get_param("Contact rate") /
436  static_cast< epiworld_double > (this->get_entity(i).size());
437 
438 
439  // Possibly correcting for a small number of agents
440  if (adjusted_contact_rate[i] > 1.0)
441  adjusted_contact_rate[i] = 1.0;
442 
443  }
444 
445  this->m_update_infected_list();
446 
447  // Setting up the quarantine parameters
448  quarantine_willingness.resize(this->size(), false);
449  isolation_willingness.resize(this->size(), false);
450  for (size_t idx = 0; idx < quarantine_willingness.size(); ++idx)
451  {
452  quarantine_willingness[idx] =
453  Model<TSeq>::runif() < this->par("Quarantine willingness");
454  isolation_willingness[idx] =
455  Model<TSeq>::runif() < this->par("Isolation willingness");
456  }
457 
458  agent_quarantine_triggered.resize(this->size(), 0u);
459  std::fill(
460  agent_quarantine_triggered.begin(),
461  agent_quarantine_triggered.end(),
462  0u
463  );
464 
465  day_flagged.resize(this->size(), 0);
466  std::fill(
467  day_flagged.begin(),
468  day_flagged.end(),
469  0
470  );
471 
472  day_onset.resize(this->size(), 0);
473  std::fill(
474  day_onset.begin(),
475  day_onset.end(),
476  0
477  );
478 
479  day_exposed.resize(this->size(), 0);
480  std::fill(
481  day_exposed.begin(),
482  day_exposed.end(),
483  0
484  );
485 
486  // Tracking matrix
487  tracking_matrix.resize(EPI_MAX_TRACKING * Model<TSeq>::size(), 0u);
488  std::fill(tracking_matrix.begin(), tracking_matrix.end(), 0u);
489 
490  tracking_matrix_size.resize(Model<TSeq>::size(), 0u);
491  std::fill(tracking_matrix_size.begin(), tracking_matrix_size.end(), 0u);
492 
493  return;
494

517 template<typename TSeq>
518 inline void ModelSEIRMixingQuarantine<TSeq>::m_update_susceptible(
519  Agent<TSeq> * p, Model<TSeq> * m
520 ) {
521 
522  if (p->get_n_entities() == 0)
523  return;
524 
525  // Downcasting to retrieve the sampler attached to the
526  // class
527  GET_MODEL(m, m_down);
528 
529  size_t ndraws = m_down->sample_agents(p, m_down->sampled_agents);
530 
531  #ifdef EPI_DEBUG
532  m_down->sampled_sizes.push_back(static_cast<int>(ndraws));
533  #endif
534 
535  if (ndraws == 0u)
536  return;
537 
538  // Drawing from the set
539  int nviruses_tmp = 0;
540  for (size_t n = 0u; n < ndraws; ++n)
541  {
542 
543  auto & neighbor = m->get_agent(m_down->sampled_agents[n]);
544 
545  auto & v = neighbor.get_virus();
546 
547  #ifdef EPI_DEBUG
548  if (nviruses_tmp >= static_cast<int>(m->array_virus_tmp.size()))
549  throw std::logic_error(
550  "Trying to add an extra element to a temporal array outside of the range."
551  );
552  #endif
553 
554  // Adding the current agent to the tracked interactions
555  m_down->m_add_tracking(neighbor.get_id(), p->get_id());
556 
557  /* And it is a function of susceptibility_reduction as well */
558  m->array_double_tmp[nviruses_tmp] =
559  (1.0 - p->get_susceptibility_reduction(v, m)) *
560  v->get_prob_infecting(m) *
561  (1.0 - neighbor.get_transmission_reduction(v, m))
562  ;
563 
564  m->array_virus_tmp[nviruses_tmp++] = &(*v);
565 
566  }
567 
568  // Running the roulette
569  int which = roulette(nviruses_tmp, m);
570 
571  if (which < 0)
572  return;
573 
574  p->set_virus(
575  *m->array_virus_tmp[which],
576  m,
577  ModelSEIRMixingQuarantine<TSeq>::EXPOSED
578  );
579 
580  return;
581 
582 };

609 template<typename TSeq>
610 inline void ModelSEIRMixingQuarantine<TSeq>::m_update_infected(
611  Agent<TSeq> * p, Model<TSeq> * m
612 ) {
613 
614  GET_MODEL(m, model);
615 
616  // Sampling whether the agent is detected or not
617  bool detected = m->runif() < 1.0/m->par("Days undetected");
618 
619  // If detected and the entity can quarantine, we start
620  // the quarantine process
621  if (detected)
622  {
623  model->agent_quarantine_triggered[p->get_id()] = 1u;
624  }
625 
626  // Checking if the agent is willing to isolate individually
627  // This is separate from quarantine and can happen even if agent cannot quarantine
628  bool isolation_detected = (m->par("Isolation period") >= 0) &&
629  detected &&
630  (model->isolation_willingness[p->get_id()])
631  ;
632 
633  // Recording the date of detection
634  if (isolation_detected)
635  model->day_flagged[p->get_id()] = m->today();
636 
637  // Computing probabilities for state change
638  const auto & v = p->get_virus();
639  m->array_double_tmp[0] = 1.0 - (1.0 - v->get_prob_recovery(m)) *
640  (1.0 - p->get_recovery_enhancer(v, m));
641  m->array_double_tmp[1] = m->par("Hospitalization rate");
642 
643  SAMPLE_FROM_PROBS(2, which);
644 
645  if (which == 0) // Recovers
646  {
647  if (isolation_detected)
648  {
649  p->change_state(
650  m, ModelSEIRMixingQuarantine<TSeq>::ISOLATED_RECOVERED
651  );
652  }
653  else
654  {
655  p->rm_virus(
656  m, ModelSEIRMixingQuarantine<TSeq>::RECOVERED
657  );
658  }
659 
660  return;
661  }
662  else if (which == 1) // Hospitalized
663  {
664 
665  if (detected)
666  {
667  p->change_state(
668  m, ModelSEIRMixingQuarantine<TSeq>::DETECTED_HOSPITALIZED
669  );
670  }
671  else
672  {
673  p->change_state(
674  m, ModelSEIRMixingQuarantine<TSeq>::HOSPITALIZED
675  );
676  }
677 
678  }
679  else if ((which == 2) && isolation_detected) // Nothing, but detected
680  {
681  // If the agent is detected, it goes to isolation
682  p->change_state(
683  m, ModelSEIRMixingQuarantine<TSeq>::ISOLATED
684  );
685 
686  }
687 
688  return ;

694  Agent<TSeq> * p, Model<TSeq> * m
695 ) {
696

698 
699  // Figuring out if the agent can be released from isolation

702

705  true: false;

714

721  {
722  p->rm_virus(
723  m,
724  ModelSEIRMixingQuarantine<TSeq>::RECOVERED
725  );
726  }

728  p->rm_virus(
729  m, ModelSEIRMixingQuarantine<TSeq>::ISOLATED_RECOVERED
730  );
731  }

862 template<typename TSeq>
863 inline void ModelSEIRMixingQuarantine<TSeq>::m_quarantine_process() {
864 
865  // Process entity-level quarantine
866  for (size_t agent_i = 0u; agent_i < Model<TSeq>::size(); ++agent_i)
867  {
868 
869  // Checking if the quarantine in the agent was triggered
870  // or not
871  if (agent_quarantine_triggered[agent_i] != 1u)
872  continue;
873 
874  if (this->par("Quarantine period") < 0)
875  continue;
876 
877  // Getting the number of contacts, if it is greater
878  // than the maximum, it means that we overflowed, so
879  // we will only quarantine the first EPI_MAX_TRACKING
880  size_t n_contacts = this->tracking_matrix_size[agent_i];
881  if (n_contacts >= EPI_MAX_TRACKING)
882  n_contacts = EPI_MAX_TRACKING;
883 
884  for (size_t contact_i = 0u; contact_i < n_contacts; ++contact_i)
885  {
886 
887  // Checking if we will detect the contact
888  if (Model<TSeq>::runif() > Model<TSeq>::par("Contact tracing success rate"))
889  continue;
890 
891  size_t contact_id = this->tracking_matrix[
892  MM(agent_i, contact_i, Model<TSeq>::size())
893  ];
894 
895  auto & agent = Model<TSeq>::get_agent(contact_id);
896 
897  if (agent.get_state() > INFECTED)
898  continue;
899 
900  // Agents with some tool won't be quarantined
901  if (agent.get_n_tools() != 0u)
902  continue;
903 
904  if (
905  quarantine_willingness[contact_id] &&
906  (Model<TSeq>::par("Quarantine period") >= 0))
907  {
908 
909  switch (agent.get_state())
910  {
911  case SUSCEPTIBLE:
912  agent.change_state(this, QUARANTINED_SUSCEPTIBLE);
913  day_flagged[contact_id] = Model<TSeq>::today();
914  break;
915  case EXPOSED:
916  agent.change_state(this, QUARANTINED_EXPOSED);
917  day_flagged[contact_id] = Model<TSeq>::today();
918  break;
919  case INFECTED:
920  if (isolation_willingness[contact_id])
921  {
922  agent.change_state(this, ISOLATED);
923  day_flagged[contact_id] = Model<TSeq>::today();
924  }
925  break;
926  default:
927  throw std::logic_error(
928  "The agent is not in a state that can be quarantined."
929  );
930  }
931 
932  }
933  }
934 
935  // Setting the quarantine process off
936  agent_quarantine_triggered[agent_i] = 2u;
937  }
938 
939  return;
940 }
