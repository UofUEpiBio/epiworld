958 template<typename TSeq>
959 inline void ModelSEIRMixingQuarantine<TSeq>::m_quarantine_process() {
960 
961  // Process entity-level quarantine
962  for (size_t agent_i = 0u; agent_i < Model<TSeq>::size(); ++agent_i)
963  {
964 
965  // Checking if the quarantine in the agent was triggered
966  // or not
967  if (
968  agent_quarantine_triggered[agent_i] !=
969  ModelSEIRMixingQuarantine<TSeq>::QUARANTINE_PROCESS_ACTIVE
970  )
971  continue;
972 
973  if (this->par("Quarantine period") < 0)
974  continue;
975 
976  // Getting the number of contacts, if it is greater
977  // than the maximum, it means that we overflowed, so
978  // we will only quarantine the first EPI_MAX_TRACKING
979  size_t n_contacts = this->tracking_matrix_size[agent_i];
980  if (n_contacts >= EPI_MAX_TRACKING)
981  n_contacts = EPI_MAX_TRACKING;
982 
983  auto success_rate = this->par("Contact tracing success rate");
984  for (size_t contact_i = 0u; contact_i < n_contacts; ++contact_i)
985  {
986 
987  // Checking if we will detect the contact
988  if (Model<TSeq>::runif() > success_rate)
989  continue;
990 
991  size_t contact_id = this->tracking_matrix[
992  MM(agent_i, contact_i, Model<TSeq>::size())
993  ];
994 
995  auto & agent = Model<TSeq>::get_agent(contact_id);
996 
997  if (agent.get_state() > INFECTED)
998  continue;
999 
1000  // Agents with some tool won't be quarantined
1001  if (agent.get_n_tools() != 0u)
1002  continue;
1003 
1004  if (quarantine_willingness[contact_id])
1005  {
1006 
1007  switch (agent.get_state())
1008  {
1009  case SUSCEPTIBLE:
1010  agent.change_state(this, QUARANTINED_SUSCEPTIBLE);
1011  day_flagged[contact_id] = Model<TSeq>::today();
1012  break;
1013  case EXPOSED:
1014  agent.change_state(this, QUARANTINED_EXPOSED);
1015  day_flagged[contact_id] = Model<TSeq>::today();
1016  break;
1017  case INFECTED:
1018  if (isolation_willingness[contact_id])
1019  {
1020  agent.change_state(this, ISOLATED);
1021  day_flagged[contact_id] = Model<TSeq>::today();
1022  }
1023  break;
1024  default:
1025  throw std::logic_error(
1026  "The agent is not in a state that can be quarantined."
1027  );
1028  }
1029 
1030  }
1031  }
1032 
1033  // Setting the quarantine process off
1034  agent_quarantine_triggered[agent_i] = QUARANTINE_PROCESS_DONE;
1035  }
1036 
1037  return;
1038 }
