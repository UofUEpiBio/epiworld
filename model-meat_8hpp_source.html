203 template<typename TSeq>
204 inline void Model<TSeq>::events_run()
205 {
206  // Making the call
207  size_t nevents_tmp = 0;
208  while (nevents_tmp < nactions)
209  {
210 
211  Event<TSeq> & a = events[nevents_tmp++];
212  Agent<TSeq> * p = a.agent;
213 
214  #ifdef EPI_DEBUG
215  if (a.new_state >= static_cast<epiworld_fast_int>(nstates))
216  throw std::range_error(
217  "The proposed state " + std::to_string(a.new_state) + " is out of range. " +
218  "The model currently has " + std::to_string(nstates - 1) + " states.");
219 
220  if (a.new_state < 0)
221  throw std::range_error(
222  "The proposed state " + std::to_string(a.new_state) + " is out of range. " +
223  "The state cannot be negative.");
224  #endif
225 
226  // Undoing the change in the transition matrix
227  if ((p->state_last_changed == today()) && (static_cast<int>(p->state) != a.new_state))
228  {
229  // Undoing state change in the transition matrix
230  // The previous state is already recorded
231  db.update_state(p->state_prev, p->state, true);
232 
233  } else
234  p->state_prev = p->state; // Recording the previous state
235 
236  // Applying function after the fact. This way, if there were
237  // updates, they can be recorded properly, before losing the information
238  p->state = a.new_state;
239  if (a.call)
240  {
241  a.call(a, this);
242  }
243 
244  // Registering that the last change was today
245  p->state_last_changed = today();
246 
247  #ifdef EPI_DEBUG
248  if (static_cast<int>(p->state) >= static_cast<int>(nstates))
249  throw std::range_error(
250  "The new state " + std::to_string(p->state) + " is out of range. " +
251  "The model currently has " + std::to_string(nstates - 1) + " states.");
252  #endif
253 
254  // Updating queue
255  if (use_queuing)
256  {
257 
258  if (a.queue == Queue<TSeq>::Everyone)
259  queue += p;
260  else if (a.queue == -Queue<TSeq>::Everyone)
261  queue -= p;
262  else if (a.queue == Queue<TSeq>::OnlySelf)
263  queue[p->get_id()]++;
264  else if (a.queue == -Queue<TSeq>::OnlySelf)
265  queue[p->get_id()]--;
266  else if (a.queue != Queue<TSeq>::NoOne)
267  throw std::logic_error(
268  "The proposed queue change is not valid. Queue values can be {-2, -1, 0, 1, 2}."
269  );
270 
271  }
272 
273  }
274 
275  // Go back to square 1
276  nactions = 0u;
277 
278  return;
279
