516  Agent<TSeq> * p, Model<TSeq> * m
517 ) {
518 
519  // If the agent has no entities, then it can't be infected
520  // (e.g., isolated agents)
521  if (p->get_n_entities() == 0)
522  return;
523 
524  // Downcasting to retrieve the sampler attached to the
525  // class
526  GET_MODEL(m, m_down);
527 
528  // Sampling infected agents
529  size_t ndraws = m_down->sample_infectious_agents(p, m_down->sampled_agents);
530 
531  #ifdef EPI_DEBUG
532  m_down->sampled_sizes.push_back(static_cast<int>(ndraws));
533  #endif
534 
535  if (ndraws == 0u)
536  return;
537 
538  // Drawing from the set
539  int nviruses_tmp = 0;
540  for (size_t n = 0u; n < ndraws; ++n)
541  {
542 
543  // Getting the neighbor and its virus
544  auto & neighbor = m->get_agent(m_down->sampled_agents[n]);
545  auto & v = neighbor.get_virus();
546 
547  #ifdef EPI_DEBUG
548  if (nviruses_tmp >= static_cast<int>(m->array_virus_tmp.size()))
549  throw std::logic_error(

551  );
552  #endif
553

555  // In this case, the infected neighbor is the one
556  // who interacts with the susceptible agent
557  m_down->contact_tracing.add_contact(neighbor.get_id(), p->get_id(), m->today());
558 
559  /* And it is a function of susceptibility_reduction as well */
560  m->array_double_tmp[nviruses_tmp] =

562  v->get_prob_infecting(m) *
563  (1.0 - neighbor.get_transmission_reduction(v, m))
564  ;
565 
566  m->array_virus_tmp[nviruses_tmp++] = &(*v);
567 
568  }

571  int which = roulette(nviruses_tmp, m);
572 
573  if (which < 0)

911  {
912

915  if (

928 
929 
930  }

937

939  {
940 
941  // Getting the location in the matrix
942  auto [contact_id, contact_date] = contact_tracing.get_contact(
943  agent_i_idx, contact_j_idx
944  );

946  // Checked agents should not be considered
947  if (!can_quarantine[contact_id])
948  continue;

950  // Will we detect the contact?
951  if (
952  Model<TSeq>::runif() >
953  Model<TSeq>::par("Contact tracing success rate")
954  )
955  continue;
956 
957  // How many days since they contacted relative to the rash onset?
958  // (recall that, in this model, the rash period is not infectious)
959  double days_since_contact =
960  static_cast<double>(this->day_rash_onset[agent_i_idx]) -
961  static_cast<double>(contact_date);

965  continue;

974  quarantine_risk_level[contact_id] = RISK_MEDIUM;
975 
976  }

1018  // quarantined should be in RISK_HIGH

1034  {
1035

1038  agent_i_idx, contact_j_idx
1039  );

1042  auto pair = std::make_pair(agent_i_idx, contact_id);
1043  if (!success_contact[pair])

1065  auto state = agent_i.get_state();

1070  (state != EXPOSED) &&
1071  (state != PRODROMAL) &&
1072  (state != RASH)

1075

1077  if (agent_i.get_n_tools() != 0u)
1078  continue;
1079

1083  );

1085  }
1086  }

1094  auto state = agent_i.get_state();
1095 
1096  // If has a tool, then skip (is vaxxed)
1097  if (agent_i.get_n_tools() != 0u)
1098  continue;

1101  // contact tracing
1102  if (agent_i.get_n_entities() != 0u)
1103  {
1104  size_t group_id = agent_i.get_entity(0u).get_id();
1105  if (groups_ids.find(group_id) != groups_ids.end())
1106  continue;
1107  }
1108 
1109  if (quarantine_risk_level[agent_i_idx] != RISK_MEDIUM)

1112  );
1113  }

1118  n_agents_per_group[i]++;
1119 
1120  #endif
1121

1123  // who triggered contact tracing
1124  Model<TSeq>::events_run();
1125  agents_triggered_contact_tracing_size = 0u;
