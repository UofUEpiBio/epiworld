525  Agent<TSeq> * p, Model<TSeq> * m
526 ) {

529  // (e.g., isolated agents)
530  if (p->get_n_entities() == 0)
531  return;

534  // class
535  GET_MODEL(m, m_down);

560  );
561  #endif
562

566  m_down->m_add_tracking(neighbor.get_id(), p->get_id());
567 
568  /* And it is a function of susceptibility_reduction as well */

572  (1.0 - neighbor.get_transmission_reduction(v, m))
573  ;
574 
575  m->array_virus_tmp[nviruses_tmp++] = &(*v);
576 
577  }

581

866  if (agents_triggered_contact_tracing_size == 0u)
867  return;

870  bool quarantine_enabled = false;

872  Model<TSeq>::size(), false
873  );

878  for (size_t i = 0u; i < agents_triggered_contact_tracing_size; ++i)
879  {
880 
881  size_t agent_i_idx = agents_triggered_contact_tracing[i];
882  auto & agent_i = Model<TSeq>::get_agent(agent_i_idx);
883

885  // It doesn't matter if the agent doesn't have a tool; we will
886  // check that later
887  if (agent_i.get_n_entities() != 0u)
888  {
889  for (auto & agent_j_idx: agent_i.get_entity(0))
890  quarantine_risk_level[agent_j_idx] = RISK_HIGH;
891  }
892 
893  // (B) Checking the contacts
894  size_t n_contacts = tracking_matrix_size[agent_i_idx];
895  if (n_contacts >= EPI_MAX_TRACKING)
896  n_contacts = EPI_MAX_TRACKING;
897 
898  for (size_t contact_j_idx = 0u; contact_j_idx < n_contacts; ++contact_j_idx)
899  {
900 
901  // Getting the location in the matrix
902  size_t loc = MM(agent_i_idx, contact_j_idx, Model<TSeq>::size());
903  size_t contact_id = this->tracking_matrix[loc];
904 
905  // Will we detect the contact?
906  if (
907  Model<TSeq>::runif() >
908  Model<TSeq>::par("Contact tracing success rate")
909  )
910  continue;
911 
912  // When did we have contact?
913  double days_since_contact =
914  static_cast<double>(this->day_rash_onset[agent_i_idx]) -
915  static_cast<double>(this->tracking_matrix_date[loc]);
916

918  if (days_since_contact > param_days_prior)
919  continue;
920 
921  // Setting the risk level to medium if not already high
922  if (quarantine_risk_level[contact_id] < RISK_HIGH)
923  quarantine_risk_level[contact_id] = RISK_MEDIUM;
924 
925  }
926 
927  }
928 
929  // Proceeding to quarantine/isolating agents
930  for (auto & agent: Model<TSeq>::get_agents())
931  {
932 
933  // If has a tool, then skip (is vaxxed)
934  if (agent.get_n_tools() != 0u)
935  continue;
936 
937  switch (agent.get_state())
938  {
939  case SUSCEPTIBLE:
940  if (quarantine_willingness[agent.get_id()])
941  {
942  agent.change_state(this, QUARANTINED_SUSCEPTIBLE);
943  day_flagged[agent.get_id()] = Model<TSeq>::today();
944  }
945  break;
946  case EXPOSED:
947  if (quarantine_willingness[agent.get_id()])
948  {
949  agent.change_state(this, QUARANTINED_EXPOSED);
950  day_flagged[agent.get_id()] = Model<TSeq>::today();
951  }
952  break;
953  case PRODROMAL:
954  if (quarantine_willingness[agent.get_id()])
955  {
956  agent.change_state(this, QUARANTINED_PRODROMAL);
957  day_flagged[agent.get_id()] = Model<TSeq>::today();
958  }
959  break;
960  case RASH:
961  if (isolation_willingness[agent.get_id()])
962  {
963  agent.change_state(this, ISOLATED);
964  day_flagged[agent.get_id()] = Model<TSeq>::today();
965  }
966  break;
967  default:
968  break;
969  }
970 
971  }
972 
973  // Applying the changes and resetting the number of agents
974  // who triggered contact tracing
975  Model<TSeq>::events_run();
976  agents_triggered_contact_tracing_size = 0u;
977 
978  return;
