504 template<typename TSeq>
505 inline void ModelMeaslesEntityQuarantine<TSeq>::m_quarantine_process() {
506 
507  // Process entity-level quarantine
508  for (size_t entity_i = 0u; entity_i < Model<TSeq>::get_entities().size(); ++entity_i)
509  {
510 
511  // Checking if the quarantine in the entity was triggered
512  // or not. We only start quarantine process if active
513  if (
514  entity_quarantine_triggered[entity_i] !=
515  ModelMeaslesEntityQuarantine<TSeq>::QUARANTINE_PROCESS_ACTIVE
516  )
517  continue;
518 
519  // Get quarantine duration for this entity based on vaccination prevalence
520  int quarantine_duration = get_quarantine_duration_for_entity(entity_i);
521 
522  if (quarantine_duration == 0)
523  continue;
524 
525  // Quarantine all eligible agents in this entity
526  auto & agents = Model<TSeq>::get_agents();
527 
528  for (auto & agent : agents)
529  {
530  if (agent.get_n_entities() > 0u)
531  {
532  const auto & entity = agent.get_entity(0u);
533  if (entity.get_id() == static_cast<int>(entity_i))
534  {
535  // Check if agent is in a state that can be quarantined
536  if (agent.get_state() > RASH)
537  continue;
538 
539  // Agents with some tool (vaccinated) won't be quarantined
540  if (agent.get_n_tools() != 0u)
541  continue;
542 
543  if (quarantine_willingness[agent.get_id()])
544  {
545 
546  switch (agent.get_state())
547  {
548  case SUSCEPTIBLE:
549  agent.change_state(this, QUARANTINED_SUSCEPTIBLE);
550  day_flagged[agent.get_id()] = Model<TSeq>::today();
551  break;
552  case EXPOSED:
553  agent.change_state(this, QUARANTINED_EXPOSED);
554  day_flagged[agent.get_id()] = Model<TSeq>::today();
555  break;
556  case PRODROMAL:
557  agent.change_state(this, QUARANTINED_PRODROMAL);
558  day_flagged[agent.get_id()] = Model<TSeq>::today();
559  break;
560  case RASH:
561  if (isolation_willingness[agent.get_id()])
562  {
563  agent.change_state(this, ISOLATED);
564  day_flagged[agent.get_id()] = Model<TSeq>::today();
565  }
566  break;
567  default:
568  throw std::logic_error(
569  "The agent is not in a state that can be quarantined."
570  );
571  }
572 
573  }
574  }
575  }
576  }
577 
578  // Setting the quarantine process off
579  entity_quarantine_triggered[entity_i] =
580  ModelMeaslesEntityQuarantine<TSeq>::QUARANTINE_PROCESS_DONE;
581  }
582 
583  return;

589 ) {

593  // Sampling how many contacts occur (this is for mixing)
594  model->sampled_agents.clear();
595  model->sample_agents(p, model->sampled_agents);
596 
597  // This computes the prob of getting any neighbor variant
598  int nviruses_tmp = 0;
599 
600  for (auto & neighbor_id: model->sampled_agents)
601  {
602 
603  auto & neighbor = m->get_agent(neighbor_id);
604  auto & v = neighbor.get_virus();
605 
606  /* \
607  * For each virus, we compute the probability of contagion
608  * and sample accordingly
609  */
610  if (v != nullptr)
611  {
612 
613  #ifdef EPI_DEBUG
614  if (nviruses_tmp >= static_cast<int>(m->array_virus_tmp.size()))
615  throw std::logic_error(
616  "Trying to add an extra element to a temporal array outside of the range."
617  );
618  #endif
619 
620  /* And it is a function of susceptibility_reduction as well */
621  m->array_double_tmp[nviruses_tmp] =
622  (1.0 - p->get_susceptibility_reduction(v, m)) *
623  v->get_prob_infecting(m) *
624  (1.0 - neighbor.get_transmission_reduction(v, m))
625  ;
626 
627  m->array_virus_tmp[nviruses_tmp++] = v.get();
628 
629  }
630 
631  }

638  SAMPLE_FROM_PROBS(nviruses_tmp, which);
639

641  if (m->runif() < m->array_double_tmp[which])
642  {

646  m,
