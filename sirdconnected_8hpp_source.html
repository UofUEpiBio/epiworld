157  );
158 
159  /* There is a bug in which runif() returns 1.0. It is rare, but
160  * we saw it here. See the Notes section in the C++ manual
161  * https://en.cppreference.com/mwiki/index.php?title=cpp/numeric/random/uniform_real_distribution&oldid=133329
162  * And the reported bug in GCC:

164  *
165  */
166  if (which == static_cast<int>(m->size()))

168 
169  // Can't sample itself
170  if (which == static_cast<int>(p->get_id()))
171  continue;
172 
173  // If the neighbor is infected, then proceed
174  auto & neighbor = m->get_agents()[which];
175  if (neighbor.get_state() == ModelSIRDCONN<TSeq>::INFECTED)
176  {

179 
180  #ifdef EPI_DEBUG

182  throw std::logic_error("Trying to add an extra element to a temporal array outside of the range.");
183  #endif
184

186  m->array_double_tmp[nviruses_tmp] =
187  (1.0 - p->get_susceptibility_reduction(v, m)) *
188  v->get_prob_infecting(m) *
189  (1.0 - neighbor.get_transmission_reduction(v, m))

193 
194  }
