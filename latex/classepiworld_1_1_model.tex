\doxysubsection*{Setting and accessing parameters from the model}
\label{_amgrp61207d51bcc1b661804b11cec5dc845e}%
 \mbox{\hyperlink{classepiworld_1_1_tools}{Tools}} can incorporate parameters included in the model. \DoxyCodeLine{[name of parameter 2]: [value in double]}
\DoxyCodeLine{...}

\end{DoxyCode}
